044import static org.junit.Assume.assumeFalse;

046
047public class TestTillerInstaller {
048
049  public TestTillerInstaller() {
050    super();
051  }
052
053  @Test
054  public void testTillerInstallation() throws IOException {
055    assumeFalse(Boolean.getBoolean("skipInstallationTest"));
056    final TillerInstaller installer = new TillerInstaller();
057    installer.init(true, 10000L);
058  }
059
060  @Test
061  public void testGetRelease() throws IOException {
062    final String releaseName = System.getProperty("testGetRelease.releaseName");
063    assumeNotNull(releaseName);
064    assumeFalse(releaseName.isEmpty());
065    try (final DefaultKubernetesClient client = new DefaultKubernetesClient();
066         final Tiller tiller = new Tiller(client)) {
067    
068      final GetHistoryRequest.Builder builder = GetHistoryRequest.newBuilder();
069      assertNotNull(builder);
070      builder.setMax(1);
071      builder.setName(releaseName);
072      final GetHistoryRequest request = builder.build();
073      assertNotNull(request);
074      assertEquals(releaseName, request.getName());
075
076      final ReleaseServiceGrpc.ReleaseServiceBlockingStub stub = tiller.getReleaseServiceBlockingStub();
077      assertNotNull(stub);
078
079      final GetHistoryResponse response = stub.getHistory(request);
080      assertNotNull(response);
081      assertEquals(1, response.getReleasesCount());
082      final List<? extends Release> releasesList = response.getReleasesList();
083      assertNotNull(releasesList);
084      assertEquals(1, releasesList.size());
085      final Release release = releasesList.get(0);
086      assertNotNull(release);
087      assertEquals(releaseName, release.getName());
088      final Chart chart = release.getChart();
089      assertNotNull(chart);
090      final Config values = chart.getValues();
091      assertNotNull(values);
092      System.out.println("*** values (raw): " + values.getRaw());
093      System.out.println("*** values (value map): " + values.getValuesMap());
094      final List<Template> templates = chart.getTemplatesList();
095      assertNotNull(templates);
096      for (final Template template : templates) {
097        assertNotNull(template);
098        System.out.println("*** retrieved template: " + template);
099      }
100      final List<Any> files = chart.getFilesList();
101      assertNotNull(files);
102      for (final Any file : files) {
103        assertNotNull(file);
104        System.out.println("*** retrieved: " + file.getTypeUrl());
105        final ByteString value = file.getValue();
106        assertNotNull(value);
107        assertTrue(value.isValidUtf8());
108      }
109    }       
110  }
