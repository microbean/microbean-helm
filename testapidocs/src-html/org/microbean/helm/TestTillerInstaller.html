090    installer.init(true);
091  }
092
093  @Test
094  public void testGetRelease() throws IOException {
095    final String releaseName = System.getProperty("testGetRelease.releaseName");
096    assumeNotNull(releaseName);
097    assumeFalse(releaseName.isEmpty());
098    try (final DefaultKubernetesClient client = new DefaultKubernetesClient();
099         final Tiller tiller = new Tiller(client)) {
100    
101      final GetHistoryRequest.Builder builder = GetHistoryRequest.newBuilder();
102      assertNotNull(builder);
103      builder.setMax(1);
104      builder.setName(releaseName);
105      final GetHistoryRequest request = builder.build();
106      assertNotNull(request);
107      assertEquals(releaseName, request.getName());
108
109      final ReleaseServiceGrpc.ReleaseServiceBlockingStub stub = tiller.getReleaseServiceBlockingStub();
110      assertNotNull(stub);
111
112      final GetHistoryResponse response = stub.getHistory(request);
113      assertNotNull(response);
114      assertEquals(1, response.getReleasesCount());
115      final List<? extends Release> releasesList = response.getReleasesList();
116      assertNotNull(releasesList);
117      assertEquals(1, releasesList.size());
118      final Release release = releasesList.get(0);
119      assertNotNull(release);
120      assertEquals(releaseName, release.getName());
121      final Chart chart = release.getChart();
122      assertNotNull(chart);
123      final Config values = chart.getValues();
124      assertNotNull(values);
125      System.out.println("*** values (raw): " + values.getRaw());
126      System.out.println("*** values (value map): " + values.getValuesMap());
127      final List<Template> templates = chart.getTemplatesList();
128      assertNotNull(templates);
129      for (final Template template : templates) {
130        assertNotNull(template);
131        System.out.println("*** retrieved template: " + template);
132      }
133      final List<Any> files = chart.getFilesList();
134      assertNotNull(files);
135      for (final Any file : files) {
136        assertNotNull(file);
137        System.out.println("*** retrieved: " + file.getTypeUrl());
138        final ByteString value = file.getValue();
139        assertNotNull(value);
140        assertTrue(value.isValidUtf8());
141      }
142    }       
143  }
