045
046public class TestChartRepositoryRepository {
047
048  public TestChartRepositoryRepository() {
049    super();
050  }
051
052  @Test
053  public void testFromYaml() throws IOException, URISyntaxException {
054    try (final InputStream inputStream = new BufferedInputStream(Thread.currentThread().getContextClassLoader().getResource("TestChartRepositoryRepository/repositories.yaml").openStream())) {
055      final ChartRepositoryRepository repositories = ChartRepositoryRepository.fromYaml(inputStream);
056      assertNotNull(repositories);
057      final Set<? extends ChartRepository> chartRepositories = repositories.getChartRepositories();
058      assertNotNull(chartRepositories);
059      assertEquals(2, chartRepositories.size());
060    }
061  }
062
063  @Test
064  public void testResolution() throws ChartResolverException, IOException, URISyntaxException {
065    final String targetDirectory = System.getProperty("project.build.directory");
066    assertNotNull(targetDirectory);
067    final Path workArea = Paths.get(targetDirectory).resolve("TestChartRepositoryRepository");
068    assertNotNull(workArea);
069    final Path indexCacheDirectory = workArea.resolve("indices");
070    assertNotNull(indexCacheDirectory);
071    Files.createDirectories(indexCacheDirectory);
072    final Path archiveCacheDirectory = workArea.resolve("archives");
073    assertNotNull(archiveCacheDirectory);
074    Files.createDirectories(archiveCacheDirectory);
075    try (final InputStream inputStream = new BufferedInputStream(Thread.currentThread().getContextClassLoader().getResource("TestChartRepositoryRepository/repositories.yaml").openStream())) {
076      final ChartRepositoryRepository repo = ChartRepositoryRepository.fromYaml(inputStream, archiveCacheDirectory, indexCacheDirectory);
077      assertNotNull(repo);
078      final Chart.Builder redis = repo.resolve("stable", "redis", "0.10.1");
079      assertNotNull(redis);
080      final MetadataOrBuilder metadata = redis.getMetadataOrBuilder();
081      assertNotNull(metadata);
082      assertEquals("redis", metadata.getName());
083      assertEquals("0.10.1", metadata.getVersion());
084    }
085  }
