017package org.microbean.helm.chart.repository;
018
019import java.io.IOException;
020
021import java.net.URI;
022import java.net.URISyntaxException;
023
024import java.nio.file.Path;
025import java.nio.file.Paths;
026
027import java.util.Map;
028import java.util.SortedSet;
029
030import org.junit.Test;
031
032import static org.junit.Assert.assertEquals;
033import static org.junit.Assert.assertNotNull;
034
035public class TestChartRepository {
036
037  public TestChartRepository() {
038    super();
039  }
040
041  @Test
042  public void testLoadIndex() throws IOException, URISyntaxException {
043    final Path indexPath = Paths.get(Thread.currentThread().getContextClassLoader().getResource("TestChartRepository/stable-index.yaml").getPath());
044    assertNotNull(indexPath);
045    final ChartRepository chartRepository = new ChartRepository("stable", new URI("https://kubernetes-charts.storage.googleapis.com/"), indexPath);
046    final ChartRepository.Index index = chartRepository.loadIndex();
047    assertNotNull(index);
048    final Map<String, SortedSet<ChartRepository.Index.Entry>> entries = index.getEntries();
049    assertNotNull(entries);
050    assertEquals(100, entries.size());
051    // Pick a couple at random and verify versions are in the proper order
052    final SortedSet<ChartRepository.Index.Entry> wordpressEntries = entries.get("wordpress");
053    assertNotNull(wordpressEntries);
054    assertEquals(25, wordpressEntries.size());
055    final ChartRepository.Index.Entry mostRecentWordpress = wordpressEntries.first();
056    assertNotNull(mostRecentWordpress);
057    assertEquals("wordpress", mostRecentWordpress.getName());
058    assertEquals("0.6.12", mostRecentWordpress.getVersion());
059  }
060
