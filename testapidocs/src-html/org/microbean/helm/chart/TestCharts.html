056
057public class TestCharts {
058
059  private URL redisUrl;
060  
061  @Before
062  public void setUp() throws IOException {
063    final URI uri = URI.create("https://kubernetes-charts.storage.googleapis.com/redis-0.10.1.tgz");
064    assertNotNull(uri);
065    final URL url = uri.toURL();
066    assertNotNull(url);
067    final URLConnection connection = url.openConnection();
068    assertNotNull(connection);
069    connection.addRequestProperty("Accept-Encoding", "gzip");
070    connection.connect();
071    assertEquals("application/x-tar", connection.getContentType());
072    this.redisUrl = url;
073  }
074
075  @After
076  public void tearDown() throws IOException {
077
078  }
079
080  @Test
081  public void testGetVersion() throws IOException {
082    assumeFalse(Boolean.getBoolean("skipClusterTests"));
083    try (final DefaultKubernetesClient client = new DefaultKubernetesClient();
084         final Tiller tiller = new Tiller(client)) {
085      final VersionOrBuilder version = tiller.getVersion();
086      assertNotNull(version);
087      assertNotNull(version.getSemVer());
088    }
089  }
090  
091  @Test
092  public void testRoundTrip() throws ExecutionException, IOException, InterruptedException {
093    assumeFalse(Boolean.getBoolean("skipClusterTests"));
094    try (final DefaultKubernetesClient client = new DefaultKubernetesClient();
095         final Tiller tiller = new Tiller(client);
096         final org.microbean.helm.ReleaseManager chartManager = new org.microbean.helm.ReleaseManager(tiller)) {
097      final InstallReleaseRequest.Builder requestBuilder = InstallReleaseRequest.newBuilder();
098      assertNotNull(requestBuilder);
099      requestBuilder.setTimeout(300L);
100      requestBuilder.setName("test-charts");
101      requestBuilder.setWait(true);
102
103      Chart.Builder chartBuilder = null;
104      try (final AbstractChartLoader<URL> loader = new URLChartLoader()) {
105        chartBuilder = loader.load(this.redisUrl);
106      }
107      assertNotNull(chartBuilder);
108      
109      final Future<InstallReleaseResponse> releaseFuture = chartManager.install(requestBuilder, chartBuilder);
110      assertNotNull(releaseFuture);
111      final Release release = releaseFuture.get().getRelease();
112      assertNotNull(release);
113      assertNotNull(release.getName());
114      assertTrue(release.hasChart());
115      assertTrue(release.hasConfig());
116
117      final UninstallReleaseRequest.Builder uninstallRequestBuilder = UninstallReleaseRequest.newBuilder();
118      assertNotNull(uninstallRequestBuilder);
119      uninstallRequestBuilder.setTimeout(300L);
120      uninstallRequestBuilder.setName("test-charts");
121      uninstallRequestBuilder.setPurge(true);
122
123      final Future<UninstallReleaseResponse> uninstallReleaseResponseFuture = chartManager.uninstall(uninstallRequestBuilder.build());
124      assertNotNull(uninstallReleaseResponseFuture);
125      final UninstallReleaseResponse response = uninstallReleaseResponseFuture.get();
126      assertNotNull(response);
127      assertNotNull(response.getRelease());
128    }
129  }
130
131}
