032import java.security.MessageDigest;

035import javax.xml.bind.DatatypeConverter;

037import org.junit.Test;
038
039import static org.junit.Assert.assertEquals;
040import static org.junit.Assert.assertNotNull;
041import static org.junit.Assert.assertTrue;
042
043public class TestDigest {
044
045  public TestDigest() {
046    super();
047  }
048
049  @Test
050  public void testDigest() throws IOException, NoSuchAlgorithmException {
051    final URL redisChart = Thread.currentThread().getContextClassLoader().getResource("redis-0.5.1/redis-0.5.1.tgz");
052    assertNotNull(redisChart);
053    final URL digest = Thread.currentThread().getContextClassLoader().getResource("redis-0.5.1/digest");
054    assertNotNull(digest);
055    assertTrue(digest.getProtocol().equals("file"));
056    String expectedDigest = null;
057    try (final BufferedReader reader = new BufferedReader(new InputStreamReader(digest.openStream(), StandardCharsets.UTF_8))) {
058      assertNotNull(reader);
059      expectedDigest = reader.readLine();
060      assertNotNull(expectedDigest);
061      expectedDigest = expectedDigest.toUpperCase();
062    }
063    final MessageDigest md = MessageDigest.getInstance("SHA-256");
064    assertNotNull(md);
065    ByteBuffer buffer = null;
066    try (final InputStream rawInputStream = redisChart.openStream();
067         final BufferedInputStream stream = rawInputStream instanceof BufferedInputStream ? (BufferedInputStream)rawInputStream : new BufferedInputStream(rawInputStream)) {
068      assertNotNull(rawInputStream);
069      assertNotNull(stream);
070      buffer = readByteBuffer(stream);
071    }
072    assertNotNull(buffer);
073    md.update(buffer);
074    assertEquals(expectedDigest, DatatypeConverter.printHexBinary(md.digest()));
075  }
076
077  private static final ByteBuffer readByteBuffer(final InputStream stream) throws IOException {
078    return ByteBuffer.wrap(read(stream));
079  }
080  
081  private static final byte[] read(final InputStream stream) throws IOException {
082    byte[] returnValue = null;
083    if (stream != null) {
084      try (final ByteArrayOutputStream buffer = new ByteArrayOutputStream()) {
085        int bytesRead;
086        final byte[] data = new byte[4096];
087        while ((bytesRead = stream.read(data, 0, data.length)) != -1) {
088          buffer.write(data, 0, bytesRead);
089        }
090        buffer.flush();
091        returnValue = buffer.toByteArray();
092      }
093    }
094    return returnValue;
095  }
096  
097}
