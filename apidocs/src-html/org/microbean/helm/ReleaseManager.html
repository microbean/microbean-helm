082  /**
083   * A {@link Pattern} specifying the constraints that a Helm release
084   * name should satisfy.
085   *
086   * <p>Because Helm release names are often used in hostnames, they
087   * should conform to <a
088   * href="https://tools.ietf.org/html/rfc1123#page-13">RFC 1123</a>.
089   * This {@link Pattern} reifies those constraints.</p>
090   *
091   * @see #validateReleaseName(String)
092   *
093   * @see <a href="https://tools.ietf.org/html/rfc1123#page-13">RFC
094   * 1123</a>
095   */
096  public static final Pattern RFC_1123_PATTERN = Pattern.compile("^[a-z0-9]([-a-z0-9]*[a-z0-9])?(\\.[a-z0-9]([-a-z0-9]*[a-z0-9])?)*$");

250  /**
251   * Installs a release.
252   *
253   * <p>This method never returns {@code null}.</p>
254   *
255   * <p>Overrides of this method must not return {@code null}.</p>
256   *
257   * @param requestBuilder the {@link InstallReleaseRequest.Builder}
258   * representing the installation request; must not be {@code null}
259   * and must {@linkplain #validate(InstallReleaseRequest.Builder)
260   * pass validation}; its {@link
261   * InstallReleaseRequest.Builder#setChart(Chart.Builder)} method
262   * will be called with the supplied {@code chartBuilder} as its
263   * argument value
264   *
265   * @param chartBuilder a {@link Chart.Builder} representing the Helm
266   * chart to install; must not be {@code null}
267   *
268   * @return a {@link Future} containing a {@link
269   * InstallReleaseResponse} that has the information requested; never
270   * {@code null}
271   *
272   * @exception NullPointerException if {@code request} is {@code
273   * null}
274   *
275   * @see org.microbean.helm.chart.AbstractChartLoader
276   */
277  public Future<InstallReleaseResponse> install(final InstallReleaseRequest.Builder requestBuilder,
278                                                final Chart.Builder chartBuilder)
279    throws IOException {
280    Objects.requireNonNull(requestBuilder);
281    Objects.requireNonNull(chartBuilder);
282    validate(requestBuilder);
283
284    // Note that the mere act of calling getValuesBuilder() has the
285    // convenient if surprising side effect of initializing the
286    // values-related innards of requestBuilder if they haven't yet
287    // been set such that, for example, requestBuilder.getValues()
288    // will no longer return null under any circumstances.  If instead
289    // here we called requestBuilder.getValues(), null *would* be
290    // returned.  For *our* code, this is fine, but Tiller's code
291    // crashes when there's a null in the values slot.
292    requestBuilder.setChart(Requirements.apply(chartBuilder, requestBuilder.getValuesBuilder()));
293    
294    String releaseNamespace = requestBuilder.getNamespace();
295    if (releaseNamespace == null || releaseNamespace.isEmpty()) {
296      final io.fabric8.kubernetes.client.Config configuration = this.getTiller().getConfiguration();
297      if (configuration == null) {
298        requestBuilder.setNamespace("default");
299      } else {
300        releaseNamespace = configuration.getNamespace();
301        if (releaseNamespace == null || releaseNamespace.isEmpty()) {
302          requestBuilder.setNamespace("default");
303        } else {
304          requestBuilder.setNamespace(releaseNamespace);
305        }
306      }
307    }
308    
309    final ReleaseServiceFutureStub stub = this.getTiller().getReleaseServiceFutureStub();
310    assert stub != null;
311    return stub.installRelease(requestBuilder.build());
312  }

619  /**
620   * Ensures that the supplied {@code name} is a valid Helm release
621   * name.
622   *
623   * <p>Because Helm release names are often used in hostnames, they
624   * should conform to <a
625   * href="https://tools.ietf.org/html/rfc1123#page-13">RFC 1123</a>.
626   * This method performs that validation by default, using the {@link
627   * #RFC_1123_PATTERN} field.</p>
628   *
629   * @param name the name to validate; may be {@code null} or
630   * {@linkplain String#isEmpty()} since Tiller will generate a valid
631   * name in such a case using the <a
632   * href="https://github.com/technosophos/moniker">{@code
633   * moniker}</a> project; if non-{@code null} must match the pattern
634   * represented by the value of the {@link #RFC_1123_PATTERN} field
635   *
636   * @see #RFC_1123_PATTERN
637   *
638   * @see <a href="https://tools.ietf.org/html/rfc1123#page-13">RFC
639   */
640  protected void validateReleaseName(final String name) {
641    if (name != null && !name.isEmpty()) {
642      final Matcher matcher = RFC_1123_PATTERN.matcher(name);
643      assert matcher != null;
644      if (!matcher.matches()) {
645        throw new IllegalArgumentException("Invalid release name: " + name + "; must match " + RFC_1123_PATTERN.toString());
646      }
647    }
648  }
