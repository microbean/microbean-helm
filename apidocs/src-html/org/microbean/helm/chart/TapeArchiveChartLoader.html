017package org.microbean.helm.chart;
018
019import java.io.IOException;
020import java.io.InputStream;
021
022import java.util.ArrayList;
023import java.util.List;
024import java.util.Map;
025
026import com.google.protobuf.Any;
027import com.google.protobuf.ByteString;
028
029import hapi.chart.ChartOuterClass.Chart;
030import hapi.chart.ConfigOuterClass.Config;
031import hapi.chart.MetadataOuterClass.Maintainer;
032import hapi.chart.MetadataOuterClass.Metadata;
033import hapi.chart.TemplateOuterClass.Template;
034
035import org.kamranzafar.jtar.TarEntry;
036import org.kamranzafar.jtar.TarInputStream;
037
038import org.yaml.snakeyaml.Yaml;
039
040public class TapeArchiveChartLoader extends StreamOrientedChartLoader<TarInputStream> {
041
042  public TapeArchiveChartLoader() {
043    super();
044  }
045  
046  @Override
047  public Chart load(final TarInputStream stream) throws IOException {
048    return this.load(stream, null, 0);
049  }
050
051  Chart load(final TarInputStream stream, TarEntry entry, final int offset) throws IOException {
052    Chart returnValue = null;
053    if (stream != null) {
054
055      final List<Template> templates = new ArrayList<>();
056      final List<Chart> subcharts = new ArrayList<>();
057      Config config = null;
058      Metadata metadata = null;
059      final List<Any> files = new ArrayList<>();
060
061      if (entry == null) {
062        entry = stream.getNextEntry();
063      }
064      while (entry != null) {
065        if (!entry.isDirectory()) {
066          String entryFullName = entry.getName();
067          if (entryFullName == null || entryFullName.isEmpty() || entryFullName.startsWith("/")) {
068            throw new IllegalArgumentException("bad chart entry: " + entryFullName);
069          }
070          if (offset > 0) {
071            entryFullName = entryFullName.substring(offset);
072          }
073          if (entryFullName == null || entryFullName.isEmpty() || entryFullName.startsWith("/")) {
074            throw new IllegalArgumentException("bad chart entry: " + entryFullName);
075          }
076          int firstSlashIndex = entryFullName.indexOf('/');
077          if (firstSlashIndex <= 0 || firstSlashIndex + 1 == entryFullName.length()) {
078            throw new IllegalArgumentException("bad chart entry: " + entryFullName);
079          }
080          final String chartName = entryFullName.substring(0, firstSlashIndex);
081          assert chartName != null;
082          assert !chartName.isEmpty();
083          final String relativeChartFile = entryFullName.substring(firstSlashIndex + 1);
084          assert relativeChartFile != null;
085          assert !relativeChartFile.isEmpty();
086          switch (relativeChartFile) {
087          case "values.yaml":
088            config = this.createConfig(stream);
089            break;
090          case "Chart.yaml":
091            metadata = this.createMetadata(stream);
092            break;
093          default:
094            if (relativeChartFile.startsWith("templates/")) {
095              if (relativeChartFile.length() > 10) { // 10 == "templates/".length()
096                templates.add(this.createTemplate(stream, relativeChartFile));
097              }
098            } else if (relativeChartFile.startsWith("charts/")) {
099              if (relativeChartFile.length() > 7) { // 7 == "charts/".length()
100                final char c = relativeChartFile.charAt(7);
101                if (c != '.' && c != '_') {
102                  final Chart subchart = this.load(stream, entry, new StringBuilder(chartName).append("/charts/").length()); // recursive
103                  if (subchart != null) {
104                    subcharts.add(subchart);
105                  }
106                }
107              }
108            } else {
109              files.add(this.createAny(stream, relativeChartFile));
110            }
111            break;
112          }
113        }
114        entry = stream.getNextEntry();
115      }
116
117      final Chart.Builder builder = Chart.newBuilder();
118      assert builder != null;
119      builder.setMetadata(metadata);
120      builder.setValues(config);
121      builder.addAllTemplates(templates);
122      builder.addAllFiles(files);
123      builder.addAllDependencies(subcharts);
124
125      returnValue = builder.build();
126    }
127    return returnValue;
128  }
129
130}
