043/**
044 * A {@link StreamOrientedChartLoader
045 * StreamOrientedChartLoader&lt;Path&gt;} that creates {@link Chart}
046 * instances from filesystem directories represented as {@link Path}
047 * objects.

049 * @author <a href="https://about.me/lairdnelson"
050 * target="_parent">Laird Nelson</a>

068    super();

077  /**
078   * Converts the supplied {@link Path}, which must be non-{@code
079   * null} and {@linkplain Files#isDirectory(Path, LinkOption...) a
080   * directory}, into an {@link Iterable} of {@link Entry} instances,
081   * each of which consists of an {@link InputStream} associated with
082   * a name.
083   *
084   * <p>This method never returns {@code null}.</p>
085   *
086   * <p>Overrides of this method are not permitted to return {@code
087   * null}.
088   *
089   * @param path the {@link Path} to read; must be non-{@code null}
090   * and must be {@linkplain Files#isDirectory(Path, LinkOption...) a
091   * directory} or an effectively empty {@link Iterable} will be
092   * returned
093   *
094   * @return a non-{@code null} {@link Iterable} of {@link Entry}
095   * instances representing named {@link InputStream}s
096   *
097   * @exception IOException if there is a problem reading from the
098   * directory represented by the supplied {@link Path} or any of its
099   * subdirectories or files
100   */
101  @Override
102  protected Iterable<? extends Entry<? extends String, ? extends InputStream>> toNamedInputStreamEntries(final Path path) throws IOException {
103    final Iterable<Entry<String, InputStream>> returnValue;
104    if (path == null || !Files.isDirectory(path)) {
105      returnValue = new EmptyIterable();
106    } else {
107      returnValue = new PathWalker(path);
108    }
109    return returnValue;
110  }
111

113  /*
114   * Inner and nested classes.
115   */

117  
118  private static final class PathWalker implements Iterable<Entry<String, InputStream>> {
119
120    private final Path directoryParent;
121
122    private final Stream<? extends Path> pathStream;
123    
124    private PathWalker(final Path directory) throws IOException {
125      super();
126      Objects.requireNonNull(directory);
127      if (!Files.isDirectory(directory)) {
128        throw new IllegalArgumentException("!Files.isDirectory(directory): " + directory);
129      }
130      final Path directoryParent = directory.getParent();
131      if (directoryParent == null) {
132        throw new IllegalArgumentException("directory.getParent() == null");
133      }
134      this.directoryParent = directoryParent;
135      final Stream<Path> pathStream;
136      final Path helmIgnore = directory.resolve(".helmIgnore");
137      assert helmIgnore != null;
138      // TODO: p in the filters below needs to be tested to see if
139      // it's, for example, foo/charts/bar/.fred--that .-prefixed
140      // directory and all of its files has to be ignored.
141      if (!Files.exists(helmIgnore)) {
142        pathStream = Files.walk(directory)
143          .filter(p -> p != null && !Files.isDirectory(p));
144      } else {
145        final HelmIgnorePathMatcher helmIgnorePathMatcher = new HelmIgnorePathMatcher(helmIgnore);
146        pathStream = Files.walk(directory)
147          .filter(p -> p != null && !Files.isDirectory(p) && !helmIgnorePathMatcher.matches(p));
148      }
149      this.pathStream = pathStream;
150    }
151
152    @Override
153    public final Iterator<Entry<String, InputStream>> iterator() {
154      return new PathIterator(this.directoryParent, this.pathStream.iterator());
155    }
156    
157  }
158
159  private static final class PathIterator implements Iterator<Entry<String, InputStream>> {
160
161    private final Path directoryParent;
162    
163    private final Iterator<? extends Path> pathIterator;
164
165    private Entry<String, InputStream> currentEntry;
166    
167    private PathIterator(final Path directoryParent, final Iterator<? extends Path> pathIterator) {
168      super();
169      Objects.requireNonNull(directoryParent);
170      Objects.requireNonNull(pathIterator);
171      if (!Files.isDirectory(directoryParent)) {
172        throw new IllegalArgumentException("!Files.isDirectory(directoryParent): " + directoryParent);
173      }
174      this.directoryParent = directoryParent;
175      this.pathIterator = pathIterator;
176    }
177
178    @Override
179    public final boolean hasNext() {
180      if (this.currentEntry != null) {
181        final InputStream oldStream = this.currentEntry.getValue();
182        if (oldStream != null) {
183          try {
184            oldStream.close();
185          } catch (final IOException ignore) {
186
187          }
188        }
189        this.currentEntry = null;
190      }      
191      return this.pathIterator != null && this.pathIterator.hasNext();
192    }
193
194    @Override
195    public final Entry<String, InputStream> next() {
196      final Path originalFile = this.pathIterator.next();
197      assert originalFile != null;
198      assert !Files.isDirectory(originalFile);
199      final Path relativeFile = this.directoryParent.relativize(originalFile);
200      assert relativeFile != null;
201      final String relativePathString = relativeFile.toString().replace('\\', '/');
202      assert relativePathString != null;
203      try {
204        this.currentEntry = new SimpleImmutableEntry<>(relativePathString, new BufferedInputStream(Files.newInputStream(originalFile)));
205      } catch (final IOException wrapMe) {
206        throw (NoSuchElementException)new NoSuchElementException(wrapMe.getMessage()).initCause(wrapMe);
207      }
208      return this.currentEntry;
209    }
210    
211  }
