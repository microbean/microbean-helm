017package org.microbean.helm.chart;
018
019import java.io.ByteArrayInputStream;
020import java.io.ByteArrayOutputStream;
021import java.io.IOException;
022import java.io.InputStream;
023
024import java.util.AbstractMap.SimpleImmutableEntry;
025import java.util.Iterator;
026import java.util.Map.Entry;
027import java.util.NoSuchElementException;
028import java.util.Objects;
029
030import java.util.zip.ZipEntry;
031import java.util.zip.ZipInputStream;
032
033public class ZipInputStreamChartLoader extends StreamOrientedChartLoader<ZipInputStream> {
034
035  public ZipInputStreamChartLoader() {
036    super();
037  }
038
039  @Override
040  protected Iterable<? extends Entry<? extends String, ? extends InputStream>> toNamedInputStreamEntries(final ZipInputStream stream) throws IOException {
041    if (stream == null) {
042      return new EmptyIterable();
043    } else {
044      return new Iterable<Entry<String, InputStream>>() {
045        @Override
046        public Iterator<Entry<String, InputStream>> iterator() {
047          return new Iterator<Entry<String, InputStream>>() {
048            private ZipEntry currentEntry;
049            
050            {
051              try {
052                this.currentEntry = stream.getNextEntry();
053              } catch (final IOException ignore) {
054                this.currentEntry = null;
055              }
056            }
057            
058            @Override
059            public boolean hasNext() {
060              return this.currentEntry != null;
061            }
062            
063            @Override
064            public Entry<String, InputStream> next() {
065              if (this.currentEntry == null) {
066                throw new NoSuchElementException();
067              }
068              ByteArrayInputStream bais = null;
069              try (final ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
070                int bytesRead = 0;
071                final byte bytes[] = new byte[4096];
072                while((bytesRead = stream.read(bytes)) >= 0) {
073                  baos.write(bytes, 0, bytesRead);
074                }
075                baos.flush();
076                bais = new ByteArrayInputStream(baos.toByteArray());
077              } catch (final IOException wrapMe) {
078                throw (NoSuchElementException)new NoSuchElementException(wrapMe.getMessage()).initCause(wrapMe);
079              }
080              final Entry<String, InputStream> returnValue = new SimpleImmutableEntry<>(this.currentEntry.getName(), bais);
081              try {
082                this.currentEntry = stream.getNextEntry();
083              } catch (final IOException ignore) {
084                this.currentEntry = null;
085              }
086              return returnValue;
087            }
088          };
089        }
090      };
091    }
092  }
093
094  /**
095   * Does nothing on purpose.
096   *
097   * @exception IOException if a subclass has overridden this method
098   * and an error occurs
099   */
100  @Override
101  public void close() throws IOException {
102
103  }
104
105  
106}
